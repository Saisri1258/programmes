class RelationalOperator
{
	public static void main(String[] args)
	{
		int a=21,b=7;
		System.out.println("a==b "+(a==b));
		System.out.println("a!=b "+(a!=b));
		System.out.println("a>b "+(a>b));
		System.out.println("a<b "+(a<b));
		System.out.println("a>=b "+(a>=b));
		System.out.println("a<=b "+(a<=b));
	}
}
